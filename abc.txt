-- Update ownership
ALTER TABLE IF EXISTS public.email_queue OWNER to rap_user;

-- Add any missing columns (e.g. attachments) in case they weren't fully synced
ALTER TABLE public.email_queue ADD COLUMN IF NOT EXISTS attachments jsonb;
ALTER TABLE public.email_queue ADD COLUMN IF NOT EXISTS status text NOT NULL DEFAULT 'QUEUED';

-- Index: ix_email_queue_consultant_id
CREATE INDEX IF NOT EXISTS ix_email_queue_consultant_id
    ON public.email_queue USING btree
    (consultant_id ASC NULLS LAST)
    TABLESPACE pg_default;

-- Index: ix_email_queue_id
CREATE INDEX IF NOT EXISTS ix_email_queue_id
    ON public.email_queue USING btree
    (id ASC NULLS LAST)
    TABLESPACE pg_default;

-- Index: ix_email_queue_requirement_id
CREATE INDEX IF NOT EXISTS ix_email_queue_requirement_id
    ON public.email_queue USING btree
    (requirement_id ASC NULLS LAST)
    TABLESPACE pg_default;

    name VARCHAR(255) NOT NULL,
    content TEXT NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- Job Matches Table
CREATE TABLE IF NOT EXISTS job_matches (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    requirement_id BIGINT NOT NULL,
    consultant_id BIGINT NOT NULL,
    match_score NUMERIC(5,2),
    match_reasoning TEXT,
    status TEXT NOT NULL DEFAULT 'PENDING',
    created_at TIMESTAMP WITH TIME ZONE NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE NOT NULL DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT fk_job_matches_requirements FOREIGN KEY (requirement_id) REFERENCES requirements (id) ON DELETE CASCADE,
    CONSTRAINT fk_job_matches_consultants FOREIGN KEY (consultant_id) REFERENCES consultants (id) ON DELETE CASCADE,
    CONSTRAINT uq_job_match_req_cons UNIQUE (requirement_id, consultant_id)
);

CREATE INDEX IF NOT EXISTS ix_job_matches_id ON job_matches (id);
CREATE INDEX IF NOT EXISTS ix_job_matches_requirement_id ON job_matches (requirement_id);
CREATE INDEX IF NOT EXISTS ix_job_matches_consultant_id ON job_matches (consultant_id);
